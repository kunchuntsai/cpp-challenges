üéØ How Circular Buffer Works:

Head (Red): Points to the oldest item to be read (POP from here)

Tail (Green): Points to where the next item will be written (PUSH to here)

When buffer is full: Push operations will fail (no overwriting)

When buffer is empty: Pop operations will fail

üìã Algorithm Steps:

1. Push Operation:
Store data at tail position ‚Üí Advance tail ‚Üí If buffer full, reject operation
2. Pop Operation:
Read data from head position ‚Üí Advance head ‚Üí Return data
3. Position Wrapping:
When reaching end of array: position = (position + 1) % buffer_size
üìù Circular Buffer Implementation v1: // =================== CIRCULAR BUFFER STRUCTURE =================== typedef struct { int* buffer; // Pointer to buffer array size_t capacity; // Maximum buffer capacity size_t head; // Read position (front of queue) size_t tail; // Write position (back of queue) size_t size; // Current number of elements } CircularBuffer; // =================== PUSH OPERATION (WRITE) =================== int circular_buffer_push(CircularBuffer* cb, int value) { // Check if buffer is full if (cb->size >= cb->capacity) return 0; // Buffer full, cannot push // Write value at tail position cb->buffer[cb->tail] = value; // Advance tail with wraparound cb->tail = (cb->tail + 1) % cb->capacity; // Increment size counter cb->size++; return 1; // Success } // =================== POP OPERATION (READ) =================== int circular_buffer_pop(CircularBuffer* cb, int* value) { // Check if buffer is empty if (cb->size == 0) return 0; // Buffer empty, cannot pop // Read value from head position *value = cb->buffer[cb->head]; // Advance head with wraparound cb->head = (cb->head + 1) % cb->capacity; // Decrement size counter cb->size--; return 1; // Success } // =================== UTILITY FUNCTIONS =================== int circular_buffer_is_empty(CircularBuffer* cb) { return cb->size == 0; } int circular_buffer_is_full(CircularBuffer* cb) { return cb->size == cb->capacity; } size_t circular_buffer_size(CircularBuffer* cb) { return cb->size; } // =================== INITIALIZATION =================== void circular_buffer_init(CircularBuffer* cb, int* buffer, size_t capacity) { cb->buffer = buffer; cb->capacity = capacity; cb->head = 0; cb->tail = 0; cb->size = 0; } // =================== USAGE EXAMPLE =================== int main() { // Static buffer allocation int buffer_array[8]; CircularBuffer cb; // Initialize buffer circular_buffer_init(&cb, buffer_array, 8); // Push some values circular_buffer_push(&cb, 10); circular_buffer_push(&cb, 20); circular_buffer_push(&cb, 30); // Pop values int value; while (circular_buffer_pop(&cb, &value)) { printf("Popped: %d\n", value); } return 0; }
